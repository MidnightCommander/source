 311 gboolean
 312 edit_drop_hotkey_menu (WDialog * h, int key)
     /*   */
 313 {
 314     int m = 0;
 315     switch (key)
 316     {
 317     case ALT ('f'):
 318         m = 0;
 319         break;
 320     case ALT ('e'):
 321         m = 1;
 322         break;
 323     case ALT ('s'):
 324         m = 2;
 325         break;
 326     case ALT ('c'):
 327         m = 3;
 328         break;
 329     case ALT ('m'):
 330         m = 4;
 331         break;
 332     case ALT ('w'):
 333         m = 5;
 334         break;
 335     case ALT ('o'):
 336         m = 6;
 337         break;
 338     default:
 339         return FALSE;
 340     }
 341 
 342     edit_drop_menu_cmd (h, m);
 343     return TRUE;
 344 }
