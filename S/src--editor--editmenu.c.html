 303 gboolean
 304 edit_drop_hotkey_menu (WDialog * h, int key)
     /*   */
 305 {
 306     int m = 0;
 307     switch (key)
 308     {
 309     case ALT ('f'):
 310         m = 0;
 311         break;
 312     case ALT ('e'):
 313         m = 1;
 314         break;
 315     case ALT ('s'):
 316         m = 2;
 317         break;
 318     case ALT ('c'):
 319         m = 3;
 320         break;
 321     case ALT ('m'):
 322         m = 4;
 323         break;
 324     case ALT ('w'):
 325         m = 5;
 326         break;
 327     case ALT ('o'):
 328         m = 6;
 329         break;
 330     default:
 331         return FALSE;
 332     }
 333 
 334     edit_drop_menu_cmd (h, m);
 335     return TRUE;
 336 }
