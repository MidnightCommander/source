  21 #define FMT_LOCALTIME(buffer, bufsize, fmt, when)               \
  22     {                                                           \
  23         struct tm *whentm;                                      \
  24         whentm = localtime(&when);                              \
  25         if (whentm == NULL)                                     \
  26         {                                                       \
  27             strncpy(buffer, INVALID_TIME_TEXT, bufsize);        \
  28             buffer[bufsize-1] = 0;                              \
  29         }                                                       \
  30         else                                                    \
  31         {                                                       \
  32             strftime(buffer, bufsize, fmt, whentm);             \
  33         }                                                       \
