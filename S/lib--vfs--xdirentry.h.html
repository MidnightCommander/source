  83 /* Single virtual file - inode */
  84 struct vfs_s_inode
  85 {
  86     struct vfs_s_super *super;  /* Archive the file is on */
  87     struct vfs_s_entry *ent;    /* Our entry in the parent directory -

  90     GQueue *subdir;             /* If this is a directory, its entry. List of vfs_s_entry */
