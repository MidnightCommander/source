 171     size_t _j;
 172

 175         const char *j = filevercmp_test_ds2[_j];
 176         int result;
 177

 179

 181             ck_assert_int_eq (! !((size_t) _i < _j), 1);
 182         else if (0 < result)
 183             ck_assert_int_eq (! !(_j < (size_t) _i), 1);
 184         else
 185             ck_assert_int_eq (! !(_j == (size_t) _i), 1);
