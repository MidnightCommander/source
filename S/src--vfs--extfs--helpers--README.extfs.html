rpm

lslR

hp48+

uc1541
 
# Break patches into chunks
patchfs

mailfs

rpms+
trpm

dpkg+

apt+
deba

audio

bpp
 
# ISO image
iso9660
 
# Amazon S3
s3+
 
# git frontend
gitfs - browse the git repo
changesetfs - list of versions of current file
patchsetfs - list of patches of current file

ulib
