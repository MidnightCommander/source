  24 /* As of May 2014, 128KiB is determined to be the minimum

  28    for i in $(seq 0 10); do

  30      printf "%7s=" $bs
  31      timeout --foreground -sINT 2 \
  32        dd bs=$bs if=/dev/zero of=/dev/null 2>&1 \
