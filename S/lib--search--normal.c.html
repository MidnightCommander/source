  49 static void
  50 mc_search__normal_translate_to_regex (GString * str)
     /*   */
  51 {
  52     gsize loop;
  53 
  54     for (loop = 0; loop < str->len; loop++)
  55         switch (str->str[loop])
  56         {
  57         case '*':
  58         case '?':
  59         case ',':
  60         case '{':
  61         case '}':
  62         case '[':
  63         case ']':
  64         case '\\':
  65         case '+':
  66         case '.':
  67         case '$':
  68         case '(':
  69         case ')':
  70         case '^':
  71         case '-':
  72         case '|':
  73             g_string_insert_c (str, loop, '\\');
  74             loop++;
  75         default:
  76             break;
  77         }
  78 }
