  48 static strtol_error_t
  49 bkm_scale (uintmax_t * x, int scale_factor)
     /*   */
  50 {
  51     if (UINTMAX_MAX / scale_factor < *x)
  52     {
  53         *x = UINTMAX_MAX;
  54         return LONGINT_OVERFLOW;
  55     }
  56 
  57     *x *= scale_factor;
  58     return LONGINT_OK;
  59 }

  63 static strtol_error_t
  64 bkm_scale_by_power (uintmax_t * x, int base, int power)
     /*   */
  65 {
  66     strtol_error_t err = LONGINT_OK;
  67     while (power-- != 0)
  68         err |= bkm_scale (x, base);
  69     return err;
  70 }

  76 strtol_error_t
  77 xstrtoumax (const char *s, char **ptr, int base, uintmax_t * val, const char *valid_suffixes)
     /*   */
  78 {
  79     char *t_ptr;
  80     char **p;
  81     uintmax_t tmp;
  82     strtol_error_t err = LONGINT_OK;
  83 
  84     g_assert (0 <= base && base <= 36);
  85 
  86     p = (ptr != NULL ? ptr : &t_ptr);
  87 
  88     {
  89         const char *q = s;
  90         unsigned char ch = *q;
  91 
  92         while (isspace (ch))
  93             ch = *++q;
  94 
  95         if (ch == '-')
  96             return LONGINT_INVALID;
  97     }
  98 
  99     errno = 0;
 100     tmp = strtol (s, p, base);
 101 
 102     if (*p == s)
 103     {
 104         /* If there is no number but there is a valid suffix, assume the
 105            number is 1.  The string is invalid otherwise.  */
 106         if (valid_suffixes != NULL && **p != '\0' && strchr (valid_suffixes, **p) != NULL)
 107             tmp = 1;
 108         else
 109             return LONGINT_INVALID;
 110     }
 111     else if (errno != 0)
 112     {
 113         if (errno != ERANGE)
 114             return LONGINT_INVALID;
 115         err = LONGINT_OVERFLOW;
 116     }
 117 
 118     /* Let valid_suffixes == NULL mean "allow any suffix".  */
 119     /* FIXME: update all callers except the ones that allow suffixes
 120        after the number, changing last parameter NULL to "".  */
 121     if (valid_suffixes == NULL)
 122     {
 123         *val = tmp;
 124         return err;
 125     }
 126 
 127     if (**p != '\0')
 128     {
 129         int suffixes = 1;
 130         strtol_error_t overflow;
 131 
 132         if (strchr (valid_suffixes, **p) == NULL)
 133         {
 134             *val = tmp;
 135             return err | LONGINT_INVALID_SUFFIX_CHAR;
 136         }
 137 
 138         base = 1024;
 139 
 140         switch (**p)
 141         {
 142         case 'E':
 143         case 'G':
 144         case 'g':
 145         case 'k':
 146         case 'K':
 147         case 'M':
 148         case 'm':
 149         case 'P':
 150         case 'T':
 151         case 't':
 152         case 'Y':
 153         case 'Z':
 154             if (strchr (valid_suffixes, '0') != NULL)
 155             {
 156                 /* The "valid suffix" '0' is a special flag meaning that
 157                    an optional second suffix is allowed, which can change
 158                    the base.  A suffix "B" (e.g. "100MB") stands for a power
 159                    of 1000, whereas a suffix "iB" (e.g. "100MiB") stands for
 160                    a power of 1024.  If no suffix (e.g. "100M"), assume
 161                    power-of-1024.  */
 162 
 163                 switch (p[0][1])
 164                 {
 165                 case 'i':
 166                     if (p[0][2] == 'B')
 167                         suffixes += 2;
 168                     break;
 169 
 170                 case 'B':
 171                 case 'D':      /* 'D' is obsolescent */
 172                     base = 1000;
 173                     suffixes++;
 174                     break;
 175                 default:
 176                     break;
 177                 }
 178             }
 179             break;
 180         default:
 181             break;
 182         }
 183 
 184         switch (**p)
 185         {
 186         case 'b':
 187             overflow = bkm_scale (&tmp, 512);
 188             break;
 189 
 190         case 'B':
 191             /* This obsolescent first suffix is distinct from the 'B'
 192                second suffix above.  E.g., 'tar -L 1000B' means change
 193                the tape after writing 1000 KiB of data.  */
 194             overflow = bkm_scale (&tmp, 1024);
 195             break;
 196 
 197         case 'c':
 198             overflow = LONGINT_OK;
 199             break;
 200 
 201         case 'E':              /* exa or exbi */
 202             overflow = bkm_scale_by_power (&tmp, base, 6);
 203             break;
 204 
 205         case 'G':              /* giga or gibi */
 206         case 'g':              /* 'g' is undocumented; for compatibility only */
 207             overflow = bkm_scale_by_power (&tmp, base, 3);
 208             break;
 209 
 210         case 'k':              /* kilo */
 211         case 'K':              /* kibi */
 212             overflow = bkm_scale_by_power (&tmp, base, 1);
 213             break;
 214 
 215         case 'M':              /* mega or mebi */
 216         case 'm':              /* 'm' is undocumented; for compatibility only */
 217             overflow = bkm_scale_by_power (&tmp, base, 2);
 218             break;
 219 
 220         case 'P':              /* peta or pebi */
 221             overflow = bkm_scale_by_power (&tmp, base, 5);
 222             break;
 223 
 224         case 'T':              /* tera or tebi */
 225         case 't':              /* 't' is undocumented; for compatibility only */
 226             overflow = bkm_scale_by_power (&tmp, base, 4);
 227             break;
 228 
 229         case 'w':
 230             overflow = bkm_scale (&tmp, 2);
 231             break;
 232 
 233         case 'Y':              /* yotta or 2**80 */
 234             overflow = bkm_scale_by_power (&tmp, base, 8);
 235             break;
 236 
 237         case 'Z':              /* zetta or 2**70 */
 238             overflow = bkm_scale_by_power (&tmp, base, 7);
 239             break;
 240 
 241         default:
 242             *val = tmp;
 243             return err | LONGINT_INVALID_SUFFIX_CHAR;
 244         }
 245 
 246         err |= overflow;
 247         *p += suffixes;
 248         if (**p != '\0')
 249             err |= LONGINT_INVALID_SUFFIX_CHAR;
 250     }
 251 
 252     *val = tmp;
 253     return err;
 254 }
